tree("Root")
	parallel
		repeat mute tree("BeAlive")
		repeat mute tree("Die")

tree("BeAlive")
	fallback
		tree("Confrontation")
		tree("ChaseEnemy")
		tree("Idle")

tree("Confrontation")
	sequence
		HasTarget
		IsTargetInRange(6.0)
		GenerateAITransition	
		fallback
			sequence
				CheckInIdleState 
				IsTargetInRange(2.5)
				CheckAITransitionInRange(0.0, 0.8)
				tree("Attack")
			sequence
				CheckInIdleState 
				not IsTargetInRange(2.0)
				CheckAITransitionInRange(0.0, 0.2)
				tree("Execute")
			sequence
				CheckAITransitionInRange(0.0, 0.75)
				tree("Surrounding")
			sequence
				IsTargetInRange(5.0)
				StopMove
				ResetAITransition
				Wait(0.5)
			sequence
				ResetAITransition
				Wait(0.5)
						
				
				


tree("ChaseEnemy")
	sequence
		HasTarget
		not IsTargetInRange(6.0)
		GenerateAITransition	
		race	
			sequence
				CheckAITransitionInRange(0.0, 0.8)
				WalkToRun
				SetInCrowd(false)
				BeginMove	
				Wait(1.5)
				ResetAITransition
			sequence
				Wait(1.0)
				CheckInIdleState 
				not IsTargetInRange(2.0)
				CheckAITransitionInRange(0.0, 0.2)
				ResetAITransition
				tree("Execute")
			

tree("Surrounding")
	sequence
		IsTargetInRange(9.0)
		RunToWalk
		SetInCrowd(true)
		ResetAITransition
		BeginMove	
		Wait(1.0)
		StopMove
	


tree("Idle")
	while
		sequence
			not HasTarget
			not FindTarget

		repeat
			Wait(1.0)


tree("Attack")
	sequence
		StopMove
		WaitRandom(0.2, 0.4)
		SetInputVectorToFaceTarget 
		ResetAITransition
		AttackCommandInput
		Wait(0.20)
		AttackCommandCancel
		GenerateAITransition	
		sequence
			CheckAITransitionInRange(0.0, 0.5)
			Wait(0.50)
			AttackCommandInput
			Wait(0.20)
			AttackCommandCancel
		ResetAITransition

tree("Execute")
	sequence
		StopMove
		SetInputVectorToFaceTarget 
		ResetAITransition
		ExecuteCommandInput
		Wait(0.20)
		ExecuteCommandCancel


tree("Die")
	IsDead

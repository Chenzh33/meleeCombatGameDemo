tree("Root")
	parallel
		repeat mute tree("BeAlive")
		repeat mute tree("Die")

tree("BeAlive")
	fallback
		tree("Confrontation")
		tree("ChaseEnemy")
		tree("Idle")

tree("Confrontation")
	sequence
		HasTarget
		IsTargetInRange(12.0)
		GenerateAITransition	
		fallback
			sequence
				CheckInIdleState 
				DodgeNotInCooldown
				IsTargetInRange(4.0)
				CheckAITransitionInRange(0.0, 0.9)
				tree("RunAway")
			sequence
				CheckInIdleState 
				IsTargetInRange(12.0)
				CheckAITransitionInRange(0.0, 0.2)
				tree("Fire")
			
			sequence
				CheckAITransitionInRange(0.0, 0.90)
				tree("Surrounding")
			sequence
				StopMove
				ResetAITransition
				Wait(0.8)
				

tree("ChaseEnemy")
	sequence
		HasTarget
		not IsTargetInRange(12.0)
		SetInCrowd(false)
		KeepMove(8.0)
			

tree("Surrounding")
	sequence
		IsTargetInRange(12.0)
		SetInCrowd(true)
		ResetAITransition
		BeginMove	
		Wait(1.0)
		StopMove
	

tree("RunAway")
	sequence
		StopMove
		SetInputVectorBackToTarget 
		ResetAITransition
		DodgeCommandInput
		Wait(0.20)
		DodgeCommandCancel

tree("Fire")
	sequence
		StopMove
		//WaitRandom(0.2, 0.4)
		SetInputVectorToFaceTarget 
		ResetAITransition
		FireCommandInput
		Wait(0.20)
		FireCommandCancel


tree("Idle")
	while
		sequence
			not HasTarget
			not FindTarget

		repeat
			Wait(1.0)


tree("Die")
	IsDead
